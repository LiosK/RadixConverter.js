 10 "use strict";
 11 
 12 /**
 13  * @constructor
 14  * @param {int|string|string[]} [ipSymbols = RadixConverter.DECIMAL_SYMBOLS]
 15  * @param {int|string|string[]} [opSymbols = RadixConverter.DECIMAL_SYMBOLS]
 16  * @throws {Error}
 17  */
 18 var RadixConverter;
 19 
 20 RadixConverter = (function() {
 21 
 22   /** @lends RadixConverter */
 23   function RadixConverter(ipSymbols, opSymbols) {
 24     var defaultSymbols = RadixConverter.DECIMAL_SYMBOLS;
 25     this.ipSymbols((ipSymbols == null) ? defaultSymbols : ipSymbols);
 26     this.opSymbols((opSymbols == null) ? defaultSymbols : opSymbols);
 27   }
 28 
 29   /**
 30    * @param {int} radix an integer ranging from 2 to 36
 31    * @returns {string}
 32    */
 33   RadixConverter.getPopularSystem = function(radix) {
 34     radix = Math.floor(radix || 0);
 35     if ((radix < 2) || (radix > 36)) {
 36       return null;
 37     } else {
 38       return ("0123456789abcdefghijklmnopqrstuvwxyz").substring(0, radix);
 39     }
 40   };
 41 
 42   /** @constant */
 43   RadixConverter.DECIMAL_SYMBOLS = RadixConverter.getPopularSystem(10);
 44 
 45   /**
 46    * @param {int|string|string[]}
 47    * @throws {Error}
 48    */
 49   RadixConverter.prototype.ipSymbols = function(x) {
 50     if (x == null) { return this._ipSymbols; }
 51 
 52     if (typeof x === "number") {
 53       x = RadixConverter.getPopularSystem(x);
 54       if (x == null) {
 55         throw new Error("invalid argument: invalid radix number for ipSymbols.");
 56       }
 57     }
 58 
 59     if (x.length > 0x8000) {
 60       throw new Error("invalid argument: too much symbols in ipSymbols.");
 61     }
 62 
 63     var ipMap = new (function NaiveAssoc() {
 64       var PREF = "assoc_";
 65       this.get = function(k) { return this[PREF + k]; };
 66       this.set = function(k, v) { this[PREF + k] = v; };
 67       this.exists = function(k) { return (PREF + k) in this; };
 68     })();
 69 
 70     for (var i = 0, len = x.length; i < len; ++i) {
 71       if (ipMap.exists(x[i])) {
 72         throw new Error("invalid argument: duplicate symbols in ipSymbols.");
 73       }
 74       ipMap.set(x[i], i);
 75     }
 76 
 77     /** @type int|string|string[] */
 78     this._ipSymbols = x;
 79     /** @type assoc */
 80     this._ipMap = ipMap;
 81 
 82     return this;
 83   };
 84 
 85   /**
 86    * @returns {int}
 87    */
 88   RadixConverter.prototype.ipRadix = function() {
 89     return this._ipSymbols.length;
 90   };
 91 
 92   /**
 93    * @param {int|string|string[]}
 94    * @throws {Error}
 95    */
 96   RadixConverter.prototype.opSymbols = function(x) {
 97     if (x == null) { return this._opSymbols; }
 98 
 99     if (typeof x === "number") {
100       x = RadixConverter.getPopularSystem(x);
101       if (x == null) {
102         throw new Error("invalid argument: invalid radix number for opSymbols.");
103       }
104     }
105 
106     if (x.length > 0x8000) {
107       throw new Error("invalid argument: too much symbols in opSymbols.");
108     }
109 
110     /** @type int|string|string[] */
111     this._opSymbols = x;
112 
113     return this;
114   };
115 
116   /**
117    * @returns {int}
118    */
119   RadixConverter.prototype.opRadix = function() {
120     return this._opSymbols.length;
121   };
122 
123   /**
124    * @param {string|string[]} ip sequence of digits (most significant first)
125    * @returns {string|string[]} sequence of digits (most significant first)
126    * @throws {Error}
127    */
128   RadixConverter.prototype.convert = function(ip) {
129     var ns = this._inputToInternal(ip), nlen = ns.length;
130     var opSymbols = this.opSymbols(), opRadix = this.opRadix();
131 
132     var op = [], i = nlen;
133     while (i > 0) {
134       var j = i, n = 0;
135       while (j--) {
136         n = (n << 16) + ns[j];
137         if (n < 0) { n += 0x100000000; }
138         ns[j] = Math.floor(n / opRadix);
139         n %= opRadix;
140       }
141       op.unshift(opSymbols[n]);
142       if (ns[i - 1] === 0) { --i; }
143     }
144 
145     return (typeof opSymbols === "string") ? op.join("") : op;
146   };
147 
148   /**
149    * @param {string|string[]} ip sequence of digits (most significant first)
150    * @returns {int16[]} sequence of 16-bit integers (least significant first)
151    * @throws {Error}
152    */
153   RadixConverter.prototype._inputToInternal = function(ip) {
154     var ipRadix = this.ipRadix();
155 
156     var ns = [0], nlen = ns.length;
157     for (var i = 0, len = ip.length; i < len; ++i) {
158       if (!this._ipMap.exists(ip[i])) {
159         throw new Error("invalid argument: unknown digit");
160       }
161 
162       var j = 0, n = this._ipMap.get(ip[i]);
163       while (j < nlen) {
164         n += ns[j] * ipRadix;
165         ns[j] = (n & 0xffff);
166         n >>>= 16;
167         if ((++j >= nlen) && (n > 0)) { ns[nlen++] = 0; }
168       }
169     }
170 
171     return ns;
172   };
173 
174   return RadixConverter;
175 
176 })(RadixConverter);
