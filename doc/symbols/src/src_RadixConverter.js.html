 12 /**
 13  * @constructor
 14  * @param {int|string|string[]} [ipSystem = 10]
 15  * @param {int|string|string[]} [opSystem = 10]
 16  * @throws {Error}
 17  */
 18 var RadixConverter;
 19 
 20 RadixConverter = (function() {
 21 
 22   /** @lends RadixConverter */
 23   function RadixConverter(ipSystem, opSystem) {
 24     this.setIpSystem((ipSystem == null) ? 10 : ipSystem);
 25     this.setOpSystem((opSystem == null) ? 10 : opSystem);
 26   }
 27 
 28   /**
 29    * Returns the popular numeral system for the given radix.
 30    *
 31    * @param {int} radix An integer ranging from 2 to 36.
 32    * @returns {string}
 33    */
 34   RadixConverter.getPopularSystem = function(radix) {
 35     radix = Math.floor(radix || 0);
 36     if ((radix < 2) || (radix > 36)) {
 37       return null;
 38     } else {
 39       return ("0123456789abcdefghijklmnopqrstuvwxyz").substring(0, radix);
 40     }
 41   };
 42 
 43   /**
 44    * Sets up the input numeral system.
 45    *
 46    * @param {int|string|string[]} sys
 47    * @returns {RadixConverter} this.
 48    * @throws {Error}
 49    */
 50   RadixConverter.prototype.setIpSystem = function(sys) {
 51     if (typeof sys === "number") {
 52       sys = RadixConverter.getPopularSystem(sys);
 53       if (sys == null) {
 54         throw new Error("invalid argument: invalid radix number for ipSystem.");
 55       }
 56     }
 57 
 58     if (sys.length < 2) {
 59       throw new Error("invalid argument: less than two symbols in ipSystem.");
 60     }
 61 
 62     if (sys.length > 0x8000) {
 63       throw new Error("invalid argument: too many symbols in ipSystem.");
 64     }
 65 
 66     var ipMap = new (function NaiveAssoc() {
 67       var PREF = "assoc_";
 68       this.get = function(k) { return this[PREF + k]; };
 69       this.set = function(k, v) { this[PREF + k] = v; };
 70       this.exists = function(k) { return (PREF + k) in this; };
 71     })();
 72 
 73     for (var i = 0, len = sys.length; i < len; ++i) {
 74       if (ipMap.exists(sys[i])) {
 75         throw new Error("invalid argument: duplicate symbols in ipSystem.");
 76       }
 77       ipMap.set(sys[i], i);
 78     }
 79 
 80     /** @type string|string[] */
 81     this._ipSymbols = sys;
 82     /** @type assoc */
 83     this._ipMap = ipMap;
 84 
 85     return this;
 86   };
 87 
 88   /**
 89    * Returns the symbol list of the input numeral system.
 90    *
 91    * @returns {string|string[]}
 92    */
 93   RadixConverter.prototype.ipSymbols = function() {
 94     return this._ipSymbols;
 95   };
 96 
 97   /**
 98    * Returns the radix of the input numeral system.
 99    *
100    * @returns {int}
101    */
102   RadixConverter.prototype.ipRadix = function() {
103     return this._ipSymbols.length;
104   };
105 
106   /**
107    * Sets up the output numeral system.
108    *
109    * @param {int|string|string[]} sys
110    * @returns {RadixConverter} this.
111    * @throws {Error}
112    */
113   RadixConverter.prototype.setOpSystem = function(sys) {
114     if (typeof sys === "number") {
115       sys = RadixConverter.getPopularSystem(sys);
116       if (sys == null) {
117         throw new Error("invalid argument: invalid radix number for opSystem.");
118       }
119     }
120 
121     if (sys.length < 2) {
122       throw new Error("invalid argument: less than two symbols in opSystem.");
123     }
124 
125     if (sys.length > 0x8000) {
126       throw new Error("invalid argument: too many symbols in opSystem.");
127     }
128 
129     /** @type string|string[] */
130     this._opSymbols = sys;
131 
132     return this;
133   };
134 
135   /**
136    * Returns the symbol list of the output numeral system.
137    *
138    * @returns {string|string[]}
139    */
140   RadixConverter.prototype.opSymbols = function() {
141     return this._opSymbols;
142   };
143 
144   /**
145    * Returns the radix of the output numeral system.
146    *
147    * @returns {int}
148    */
149   RadixConverter.prototype.opRadix = function() {
150     return this._opSymbols.length;
151   };
152 
153   /**
154    * @param {string|string[]} ip sequence of digits (most significant first)
155    * @returns {string|string[]} sequence of digits (most significant first)
156    * @throws {Error}
157    */
158   RadixConverter.prototype.convert = function(ip) {
159     var ns = this._inputToInternal(ip), nlen = ns.length;
160     var opSymbols = this.opSymbols(), opRadix = this.opRadix();
161 
162     var op = [], i = nlen;
163     while (i > 0) {
164       var j = i, n = 0;
165       while (j--) {
166         n = (n << 16) + ns[j];
167         if (n < 0) { n += 0x100000000; }
168         ns[j] = Math.floor(n / opRadix);
169         n %= opRadix;
170       }
171       op.unshift(opSymbols[n]);
172       if (ns[i - 1] === 0) { --i; }
173     }
174 
175     return (typeof opSymbols === "string") ? op.join("") : op;
176   };
177 
178   /**
179    * @param {string|string[]} ip sequence of digits (most significant first)
180    * @returns {int16[]} sequence of 16-bit integers (least significant first)
181    * @throws {Error}
182    */
183   RadixConverter.prototype._inputToInternal = function(ip) {
184     var ipRadix = this.ipRadix(), ipMap = this._ipMap;
185 
186     var ns = [0], nlen = ns.length;
187     for (var i = 0, len = ip.length; i < len; ++i) {
188       if (!ipMap.exists(ip[i])) {
189         throw new Error("invalid argument: unknown digit");
190       }
191 
192       var j = 0, n = ipMap.get(ip[i]);
193       while (j < nlen) {
194         n += ns[j] * ipRadix;
195         ns[j] = (n & 0xffff);
196         n >>>= 16;
197         if ((++j >= nlen) && (n > 0)) { ns[nlen++] = 0; }
198       }
199     }
200 
201     return ns;
202   };
203 
204   return RadixConverter;
205 
206 })(RadixConverter);
